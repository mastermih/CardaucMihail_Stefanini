SELECT 
                   op.order_id, 
                   op.quantity, 
                   op.price_product, 
                   op.product_id,  
                   op.parent_product_id, 
                   p.product_name, 
                   p.category_type, 
                   p.image_path    
                FROM 
                   order_product op 
                JOIN 
                   product p 
                ON 
                   op.product_id = p.id 
                WHERE 
                   op.order_id = 217;



SELECT o.*, oo.user_id, oo.assigned_role FROM orders o LEFT JOIN order_operators oo ON o.id = oo.order_id ORDER BY o.id ASC LIMIT 5;
