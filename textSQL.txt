SELECT 
                   op.order_id, 
                   op.quantity, 
                   op.price_product, 
                   op.product_id,  
                   op.parent_product_id, 
                   p.product_name, 
                   p.category_type, 
                   p.image_path    
                FROM 
                   order_product op 
                JOIN 
                   product p 
                ON 
                   op.product_id = p.id 
                WHERE 
                   op.order_id = 217;
